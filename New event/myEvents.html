let timeGreeting = 'Good Morning'; if (hour >= 12 && hour < 17) timeGreeting = 'Good Afternoon'; else if (hour >= 17) timeGreeting = 'Good Evening'; greeting.textContent = `${timeGreeting}, ${profile.first_name}`; // Show the most recent event renderEventDetails(latestEvent); } else { showNoEventsMessage(); } } catch (error) { console.error('Failed to load event details:', error); showNoEventsMessage(); } } function renderEventDetails(event) { const eventNameEl = document.getElementById('event-name'); const eventTitleEl = document.getElementById('event-title'); const eventDurationEl = document.getElementById('event-duration'); if (eventNameEl) eventNameEl.textContent = event.name; if (eventTitleEl) eventTitleEl.textContent = event.name; if (eventDurationEl) eventDurationEl.textContent = `${event.duration} minutes`; if (event.image) { const banner = document.getElementById('event-banner'); if (banner) { banner.src = event.image; banner.style.display = 'block'; const summaryEl = document.getElementById('summary'); if (summaryEl) summaryEl.textContent = summary; } catch (error) { const summaryEl = document.getElementById('summary'); if (summaryEl) summaryEl.textContent = 'Availability not configured'; } } function showNoEventsMessage() { const eventNameEl = document.getElementById('event-name'); const eventTitleEl = document.getElementById('event-title'); const eventDurationEl = document.getElementById('event-duration'); const summaryEl = document.getElementById('summary'); if (eventNameEl) eventNameEl.textContent = 'No Events'; if (eventTitleEl) eventTitleEl.textContent = 'No Events Created'; if (eventDurationEl) eventDurationEl.textContent = 'N/A'; } } function renderMeetings(meetings) { const meetingList = document.querySelector(".meeting-list"); if (!meetingList) return; // Clear existing dynamic meetings const existingMeetings = meetingList.querySelectorAll("li.dynamic-meeting"); existingMeetings.forEach(li => li.remove()); meetings.forEach((meeting) => { const li = document.createElement("li"); li.className = "dynamic-meeting"; const meetingTime = formatDateTime(meeting.start_time); li.innerHTML = `
${meeting.title}
${meetingTime}
`; meetingList.appendChild(li); }); } function renderRecentActivity(meetings) { const tbody = document.getElementById('recent-activity-tbody'); if (!tbody) return; tbody.innerHTML = ''; meetings.slice(0, 5).forEach(meeting => { const row = document.createElement('tr'); const meetingTime = formatDateTime(meeting.start_time); row.innerHTML = ` ${meeting.title} ${meeting.invitee_name} ${meetingTime} ${meeting.location_type === 'zoom' ?
